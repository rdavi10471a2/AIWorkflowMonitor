using System.Text.Json;
using Microsoft.CodeAnalysis;
using Microsoft.CodeAnalysis.CSharp;
using Microsoft.CodeAnalysis.CSharp.Syntax;

namespace AIWorkflowMonitor;

internal static partial class AIWorkflowRunner
{
    private static readonly JsonSerializerOptions CompactJsonOptions = new()
    {
        WriteIndented = false
    };

    private static readonly HashSet<string> IncludedExtensions = new(StringComparer.OrdinalIgnoreCase)
    {
        ".cs",
        ".csproj",
        ".sln",
        ".slnx"
    };

    private static readonly HashSet<string> IgnoredDirectories = new(StringComparer.OrdinalIgnoreCase)
    {
        ".git",
        ".vs",
        "bin",
        "obj",
        "node_modules",
        "packages",
        "Working",
        "archive",
        "Archive"
    };

    private static SourceMapRefreshResult RefreshSourceMapIfNeeded(
        string observedRoot,
        string sourceMapStateDir,
        bool forceRefresh,
        string targetRelativePath)
    {
        var rootName = new DirectoryInfo(observedRoot).Name;
        var legacyWorkspaceMapPath = Path.Combine(sourceMapStateDir, $"{rootName}.workspace.sourcemap.json");
        var snapshotPath = Path.Combine(sourceMapStateDir, $"{rootName}.snapshot.state");
        var sourceMapIndexPath = Path.Combine(sourceMapStateDir, "sourcemap.index.json");

        var files = EnumerateWorkspaceFiles(observedRoot);
        var snapshotLines = BuildWorkspaceSnapshotLines(observedRoot, files);
        var snapshotText = string.Join(Environment.NewLine, snapshotLines);
        var projectRoots = GetProjectRoots(sourceMapStateDir, observedRoot, files);
        var allMapPaths = projectRoots
            .SelectMany(static p => new[] { p.WorkspaceMapPath, p.SemanticMapPath })
            .ToList();
        allMapPaths.Add(sourceMapIndexPath);

        string? previousSnapshot = null;
        var allMapFilesExist = File.Exists(snapshotPath) && allMapPaths.All(File.Exists);
        if (!forceRefresh && allMapFilesExist)
        {
            previousSnapshot = File.ReadAllText(snapshotPath);
        }

        var jsonOptions = new JsonSerializerOptions
        {
            WriteIndented = true
        };

        // Default path: refresh sourcemap for the currently read/targeted file on each run.
        // This keeps the contract index warm without forcing a full workspace rebuild.
        if (!forceRefresh
            && allMapFilesExist
            && !string.IsNullOrWhiteSpace(targetRelativePath)
            && TryRefreshSingleScopeSourceMap(
                observedRoot,
                sourceMapStateDir,
                sourceMapIndexPath,
                files,
                snapshotLines,
                projectRoots,
                targetRelativePath,
                jsonOptions))
        {
            File.WriteAllText(snapshotPath, snapshotText);
            return new SourceMapRefreshResult(true, allMapPaths, "single-file");
        }

        if (!forceRefresh
            && !string.IsNullOrWhiteSpace(previousSnapshot)
            && string.Equals(previousSnapshot, snapshotText, StringComparison.Ordinal))
        {
            return new SourceMapRefreshResult(false, allMapPaths, "unchanged");
        }

        if (!forceRefresh
            && !string.IsNullOrWhiteSpace(previousSnapshot)
            && allMapPaths.All(File.Exists)
            && TryGetChangedRelativePaths(previousSnapshot, snapshotText, out var changedPaths)
            && changedPaths.Count == 1
            && TryRefreshSingleScopeSourceMap(
                observedRoot,
                sourceMapStateDir,
                sourceMapIndexPath,
                files,
                snapshotLines,
                projectRoots,
                changedPaths[0],
                jsonOptions))
        {
            File.WriteAllText(snapshotPath, snapshotText);
            return new SourceMapRefreshResult(true, allMapPaths, "single-change");
        }

        var scopeIndexes = new List<SourceMapScopeIndexEntry>();

        foreach (var projectRoot in projectRoots)
        {
            var workspaceMap = BuildProjectWorkspaceMap(observedRoot, files, snapshotLines, projectRoot.RelativeRoot);
            var workspaceJson = JsonSerializer.Serialize(workspaceMap, jsonOptions);
            File.WriteAllText(projectRoot.WorkspaceMapPath, workspaceJson);

            var semanticMap = BuildProjectSemanticMap(observedRoot, files, projectRoot.RelativeRoot);
            var semanticJson = JsonSerializer.Serialize(semanticMap, jsonOptions);
            File.WriteAllText(projectRoot.SemanticMapPath, semanticJson);
            scopeIndexes.Add(BuildScopeIndexEntry(projectRoot, semanticMap));

            var legacyProjectMapPath = Path.Combine(sourceMapStateDir, $"{projectRoot.RelativeRoot}.sourcemap.json");
            if (File.Exists(legacyProjectMapPath))
            {
                File.Delete(legacyProjectMapPath);
            }
        }

        if (File.Exists(legacyWorkspaceMapPath))
        {
            File.Delete(legacyWorkspaceMapPath);
        }

        var sourceMapIndex = new SourceMapIndexDocument
        {
            ObservedRoot = Path.GetFullPath(observedRoot),
            GeneratedUtc = DateTime.UtcNow.ToString("O"),
            ScopeCount = scopeIndexes.Count,
            Scopes = scopeIndexes
        };
        File.WriteAllText(sourceMapIndexPath, JsonSerializer.Serialize(sourceMapIndex, jsonOptions));

        File.WriteAllText(snapshotPath, snapshotText);
        return new SourceMapRefreshResult(true, allMapPaths, forceRefresh ? "full-forced" : "full");
    }

    private static bool TryRefreshSingleScopeSourceMap(
        string observedRoot,
        string sourceMapStateDir,
        string sourceMapIndexPath,
        List<WorkspaceFileInfo> files,
        List<string> snapshotLines,
        List<ProjectRootMap> projectRoots,
        string changedRelativePath,
        JsonSerializerOptions jsonOptions)
    {
        try
        {
            if (!File.Exists(sourceMapIndexPath))
            {
                return false;
            }

            var scope = ResolveScopeForRelativePath(changedRelativePath, projectRoots);
            if (string.IsNullOrWhiteSpace(scope))
            {
                return false;
            }

            var targetRoot = projectRoots.FirstOrDefault(p => string.Equals(p.RelativeRoot, scope, StringComparison.OrdinalIgnoreCase));
            if (string.IsNullOrWhiteSpace(targetRoot.RelativeRoot))
            {
                return false;
            }

            var existingIndex = JsonSerializer.Deserialize<SourceMapIndexDocument>(File.ReadAllText(sourceMapIndexPath), CompactJsonOptions);
            if (existingIndex is null)
            {
                return false;
            }

            var expectedScopes = projectRoots
                .Select(static p => p.RelativeRoot)
                .ToHashSet(StringComparer.OrdinalIgnoreCase);
            var existingScopes = existingIndex.Scopes
                .Select(static s => s.Scope)
                .ToHashSet(StringComparer.OrdinalIgnoreCase);
            if (!expectedScopes.SetEquals(existingScopes))
            {
                // Scope set changed (new/removed project roots). Fall back to full rebuild.
                return false;
            }

            var workspaceMap = BuildProjectWorkspaceMap(observedRoot, files, snapshotLines, targetRoot.RelativeRoot);
            File.WriteAllText(targetRoot.WorkspaceMapPath, JsonSerializer.Serialize(workspaceMap, jsonOptions));

            var semanticMap = BuildProjectSemanticMap(observedRoot, files, targetRoot.RelativeRoot);
            File.WriteAllText(targetRoot.SemanticMapPath, JsonSerializer.Serialize(semanticMap, jsonOptions));

            var refreshedScope = BuildScopeIndexEntry(targetRoot, semanticMap);
            var mergedScopes = new List<SourceMapScopeIndexEntry>();
            foreach (var root in projectRoots)
            {
                if (string.Equals(root.RelativeRoot, targetRoot.RelativeRoot, StringComparison.OrdinalIgnoreCase))
                {
                    mergedScopes.Add(refreshedScope);
                    continue;
                }

                var existingScope = existingIndex.Scopes.FirstOrDefault(s => string.Equals(s.Scope, root.RelativeRoot, StringComparison.OrdinalIgnoreCase));
                if (existingScope is null)
                {
                    return false;
                }

                mergedScopes.Add(existingScope);
            }

            var updatedIndex = new SourceMapIndexDocument
            {
                ObservedRoot = Path.GetFullPath(observedRoot),
                GeneratedUtc = DateTime.UtcNow.ToString("O"),
                ScopeCount = mergedScopes.Count,
                Scopes = mergedScopes
            };
            File.WriteAllText(sourceMapIndexPath, JsonSerializer.Serialize(updatedIndex, jsonOptions));

            var legacyProjectMapPath = Path.Combine(sourceMapStateDir, $"{targetRoot.RelativeRoot}.sourcemap.json");
            if (File.Exists(legacyProjectMapPath))
            {
                File.Delete(legacyProjectMapPath);
            }

            return true;
        }
        catch
        {
            return false;
        }
    }

    private static string ResolveScopeForRelativePath(string relativePath, List<ProjectRootMap> projectRoots)
    {
        if (string.IsNullOrWhiteSpace(relativePath))
        {
            return string.Empty;
        }

        var normalized = relativePath.Replace(Path.AltDirectorySeparatorChar, Path.DirectorySeparatorChar);
        var firstSeparator = normalized.IndexOf(Path.DirectorySeparatorChar);
        var firstSegment = firstSeparator >= 0 ? normalized[..firstSeparator] : normalized;

        if (projectRoots.Any(p => string.Equals(p.RelativeRoot, firstSegment, StringComparison.OrdinalIgnoreCase)))
        {
            return firstSegment;
        }

        // Root-level project/sln scopes represent whole-workspace maps in this layout.
        var workspaceScope = projectRoots
            .Select(static p => p.RelativeRoot)
            .FirstOrDefault(static root =>
                root.EndsWith(".sln", StringComparison.OrdinalIgnoreCase)
                || root.EndsWith(".slnx", StringComparison.OrdinalIgnoreCase)
                || root.EndsWith(".csproj", StringComparison.OrdinalIgnoreCase));
        if (!string.IsNullOrWhiteSpace(workspaceScope))
        {
            return workspaceScope;
        }

        if (projectRoots.Count == 1)
        {
            return projectRoots[0].RelativeRoot;
        }

        return string.Empty;
    }

    private static bool TryGetChangedRelativePaths(string oldSnapshotText, string newSnapshotText, out List<string> changedPaths)
    {
        changedPaths = [];
        try
        {
            var oldState = ParseSnapshotState(oldSnapshotText);
            var newState = ParseSnapshotState(newSnapshotText);
            var allKeys = oldState.Keys
                .Concat(newState.Keys)
                .Distinct(StringComparer.OrdinalIgnoreCase);

            foreach (var key in allKeys)
            {
                var oldSig = oldState.TryGetValue(key, out var o) ? o : string.Empty;
                var newSig = newState.TryGetValue(key, out var n) ? n : string.Empty;
                if (!string.Equals(oldSig, newSig, StringComparison.Ordinal))
                {
                    changedPaths.Add(key);
                }
            }

            changedPaths = changedPaths
                .Distinct(StringComparer.OrdinalIgnoreCase)
                .OrderBy(static p => p, StringComparer.OrdinalIgnoreCase)
                .ToList();
            return true;
        }
        catch
        {
            changedPaths = [];
            return false;
        }
    }

    private static Dictionary<string, string> ParseSnapshotState(string snapshotText)
    {
        var result = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
        var lines = snapshotText.Split(new[] { "\r\n", "\n" }, StringSplitOptions.None);
        foreach (var line in lines)
        {
            if (string.IsNullOrWhiteSpace(line)
                || line.StartsWith("observed_root=", StringComparison.OrdinalIgnoreCase)
                || line.StartsWith("tracked_file_count=", StringComparison.OrdinalIgnoreCase))
            {
                continue;
            }

            var firstPipe = line.IndexOf('|');
            if (firstPipe <= 0 || firstPipe >= line.Length - 1)
            {
                continue;
            }

            var relativePath = line[..firstPipe];
            var signature = line[(firstPipe + 1)..];
            if (!string.IsNullOrWhiteSpace(relativePath))
            {
                result[relativePath] = signature;
            }
        }

        return result;
    }

    private static SourceMapScopeIndexEntry BuildScopeIndexEntry(ProjectRootMap projectRoot, ProjectSemanticMapDocument semanticMap)
    {
        // Flatten type-level semantic data into compact signature lists so callers can
        // perform quick member lookups from sourcemap.index.json without loading full maps.
        var methods = new List<SourceMapSignatureIndexEntry>();
        var properties = new List<SourceMapSignatureIndexEntry>();
        var fields = new List<SourceMapSignatureIndexEntry>();
        var eventsIndex = new List<SourceMapSignatureIndexEntry>();

        foreach (var sourceFile in semanticMap.SourceFiles)
        {
            foreach (var type in sourceFile.Types)
            {
                methods.AddRange(type.Methods.Select(m => new SourceMapSignatureIndexEntry
                {
                    TypeName = type.QualifiedName,
                    MemberName = m.Name,
                    Signature = m.Signature,
                    FilePath = sourceFile.RelativePath
                }));

                properties.AddRange(type.Properties.Select(p => new SourceMapSignatureIndexEntry
                {
                    TypeName = type.QualifiedName,
                    MemberName = p.Name,
                    Signature = p.Signature,
                    FilePath = sourceFile.RelativePath
                }));

                fields.AddRange(type.Fields.Select(f => new SourceMapSignatureIndexEntry
                {
                    TypeName = type.QualifiedName,
                    MemberName = f.Name,
                    Signature = f.Signature,
                    FilePath = sourceFile.RelativePath
                }));

                eventsIndex.AddRange(type.Events.Select(e => new SourceMapSignatureIndexEntry
                {
                    TypeName = type.QualifiedName,
                    MemberName = e.Name,
                    Signature = e.Signature,
                    FilePath = sourceFile.RelativePath
                }));
            }
        }

        return new SourceMapScopeIndexEntry
        {
            Scope = projectRoot.RelativeRoot,
            WorkspaceMapPath = projectRoot.WorkspaceMapPath,
            SemanticMapPath = projectRoot.SemanticMapPath,
            SourceFileCount = semanticMap.SourceFiles.Count,
            TypeCount = semanticMap.SourceFiles.Sum(static sf => sf.Types.Count),
            MethodSignatureCount = methods.Count,
            PropertySignatureCount = properties.Count,
            FieldSignatureCount = fields.Count,
            EventSignatureCount = eventsIndex.Count,
            MethodSignatures = methods,
            PropertySignatures = properties,
            FieldSignatures = fields,
            EventSignatures = eventsIndex
        };
    }

    private static List<WorkspaceFileInfo> EnumerateWorkspaceFiles(string observedRoot)
    {
        var root = Path.GetFullPath(observedRoot);
        var results = new List<WorkspaceFileInfo>();
        var pending = new Stack<string>();
        pending.Push(root);

        while (pending.Count > 0)
        {
            var current = pending.Pop();
            foreach (var dir in Directory.EnumerateDirectories(current))
            {
                var name = Path.GetFileName(dir);
                if (IgnoredDirectories.Contains(name))
                {
                    continue;
                }

                pending.Push(dir);
            }

            foreach (var file in Directory.EnumerateFiles(current))
            {
                var extension = Path.GetExtension(file);
                if (!IncludedExtensions.Contains(extension))
                {
                    continue;
                }

                var info = new FileInfo(file);
                var relative = Path.GetRelativePath(root, file);
                results.Add(new WorkspaceFileInfo(
                    file,
                    relative,
                    info.LastWriteTimeUtc.Ticks,
                    info.Exists ? info.Length : 0,
                    extension));
            }
        }

        results.Sort(static (a, b) => StringComparer.OrdinalIgnoreCase.Compare(a.RelativePath, b.RelativePath));
        return results;
    }

    private static List<string> BuildWorkspaceSnapshotLines(string observedRoot, List<WorkspaceFileInfo> files)
    {
        var lines = new List<string>
        {
            $"observed_root={Path.GetFullPath(observedRoot)}",
            $"tracked_file_count={files.Count}"
        };

        foreach (var file in files)
        {
            lines.Add($"{file.RelativePath}|{file.LastWriteUtcTicks}|{file.Length}");
        }

        return lines;
    }

    private static ProjectWorkspaceMapDocument BuildProjectWorkspaceMap(string observedRoot, List<WorkspaceFileInfo> files, List<string> snapshotLines, string projectRootRelative)
    {
        var projectScopedFiles = FilterProjectFiles(files, projectRootRelative);
        var projectFiles = projectScopedFiles
            .Where(static f => string.Equals(f.Extension, ".csproj", StringComparison.OrdinalIgnoreCase))
            .Select(static f => f.RelativePath)
            .OrderBy(static p => p, StringComparer.OrdinalIgnoreCase)
            .ToList();

        var solutionFiles = projectScopedFiles
            .Where(static f => string.Equals(f.Extension, ".sln", StringComparison.OrdinalIgnoreCase)
                || string.Equals(f.Extension, ".slnx", StringComparison.OrdinalIgnoreCase))
            .Select(static f => f.RelativePath)
            .OrderBy(static p => p, StringComparer.OrdinalIgnoreCase)
            .ToList();

        var trackedFiles = projectScopedFiles
            .Select(static f => new WorkspaceTrackedFileEntry
            {
                RelativePath = f.RelativePath,
                Extension = f.Extension,
                LastWriteUtcTicks = f.LastWriteUtcTicks,
                SizeBytes = f.Length
            })
            .ToList();

        return new ProjectWorkspaceMapDocument
        {
            ObservedRoot = Path.GetFullPath(observedRoot),
            Scope = projectRootRelative,
            GeneratedUtc = DateTime.UtcNow.ToString("O"),
            Snapshot = snapshotLines,
            ProjectFiles = projectFiles,
            SolutionFiles = solutionFiles,
            TrackedFiles = trackedFiles
        };
    }

    private static ProjectSemanticMapDocument BuildProjectSemanticMap(string observedRoot, List<WorkspaceFileInfo> files, string projectRootRelative)
    {
        var projectFiles = FilterProjectFiles(files, projectRootRelative);
        var sourceFiles = projectFiles
            .Where(static f => string.Equals(f.Extension, ".cs", StringComparison.OrdinalIgnoreCase))
            .ToList();

        var projectDefinitions = projectFiles
            .Where(static f => string.Equals(f.Extension, ".csproj", StringComparison.OrdinalIgnoreCase))
            .Select(static f => f.RelativePath)
            .OrderBy(static p => p, StringComparer.OrdinalIgnoreCase)
            .ToList();

        var solutionDefinitions = projectFiles
            .Where(static f => string.Equals(f.Extension, ".sln", StringComparison.OrdinalIgnoreCase)
                || string.Equals(f.Extension, ".slnx", StringComparison.OrdinalIgnoreCase))
            .Select(static f => f.RelativePath)
            .OrderBy(static p => p, StringComparer.OrdinalIgnoreCase)
            .ToList();

        var semanticSourceFiles = sourceFiles
            .Select(BuildSourceMapFileEntry)
            .ToList();

        return new ProjectSemanticMapDocument
        {
            ObservedRoot = Path.GetFullPath(observedRoot),
            Scope = projectRootRelative,
            GeneratedUtc = DateTime.UtcNow.ToString("O"),
            Intent = "contract-map",
            ProjectFiles = projectDefinitions,
            SolutionFiles = solutionDefinitions,
            SourceFiles = semanticSourceFiles
        };
    }

    private static List<WorkspaceFileInfo> FilterProjectFiles(List<WorkspaceFileInfo> files, string projectRootRelative)
    {
        // Project scope maps are generated by first directory segment (or root fallback).
        // This keeps each map bounded to a logical slice of the observed repository.
        var prefix = projectRootRelative + Path.DirectorySeparatorChar;
        var matched = files
            .Where(f => f.RelativePath.StartsWith(prefix, StringComparison.OrdinalIgnoreCase))
            .ToList();

        if (matched.Count == 0
            && (projectRootRelative.EndsWith(".csproj", StringComparison.OrdinalIgnoreCase)
                || projectRootRelative.EndsWith(".sln", StringComparison.OrdinalIgnoreCase)
                || projectRootRelative.EndsWith(".slnx", StringComparison.OrdinalIgnoreCase)))
        {
            return files;
        }

        return matched;
    }

    private static List<ProjectRootMap> GetProjectRoots(string sourceMapStateDir, string observedRoot, List<WorkspaceFileInfo> files)
    {
        // Group by first path segment so each repo area gets its own scope file pair:
        //   <scope>.workspace.sourcemap.json
        //   <scope>.semantic.sourcemap.json
        var projectRoots = files
            .Where(static f =>
                string.Equals(f.Extension, ".csproj", StringComparison.OrdinalIgnoreCase)
                || string.Equals(f.Extension, ".sln", StringComparison.OrdinalIgnoreCase)
                || string.Equals(f.Extension, ".slnx", StringComparison.OrdinalIgnoreCase))
            .Select(static f => f.RelativePath.Split(Path.DirectorySeparatorChar, Path.AltDirectorySeparatorChar)[0])
            .Where(static firstSegment => !string.IsNullOrWhiteSpace(firstSegment))
            .Distinct(StringComparer.OrdinalIgnoreCase)
            .OrderBy(static p => p, StringComparer.OrdinalIgnoreCase)
            .Select(p => new ProjectRootMap(
                p,
                Path.Combine(sourceMapStateDir, $"{p}.workspace.sourcemap.json"),
                Path.Combine(sourceMapStateDir, $"{p}.semantic.sourcemap.json")))
            .ToList();

        if (projectRoots.Count == 0)
        {
            var rootName = new DirectoryInfo(observedRoot).Name;
            projectRoots.Add(new ProjectRootMap(
                rootName,
                Path.Combine(sourceMapStateDir, $"{rootName}.workspace.sourcemap.json"),
                Path.Combine(sourceMapStateDir, $"{rootName}.semantic.sourcemap.json")));
        }

        return projectRoots;
    }

    private static SourceMapFileEntry BuildSourceMapFileEntry(WorkspaceFileInfo source)
    {
        var text = File.ReadAllText(source.FullPath);
        var syntaxTree = CSharpSyntaxTree.ParseText(text, path: source.FullPath);
        var root = syntaxTree.GetCompilationUnitRoot();

        var namespaces = root.DescendantNodes()
            .OfType<BaseNamespaceDeclarationSyntax>()
            .Select(static n => n.Name.ToString())
            .Distinct(StringComparer.Ordinal)
            .OrderBy(static n => n, StringComparer.Ordinal)
            .ToList();

        var types = root.DescendantNodes()
            .OfType<TypeDeclarationSyntax>()
            .Select(t => CreateTypeEntry(syntaxTree, t))
            .ToList();

        var lineCount = CountLines(text);
        var diagnostics = syntaxTree.GetDiagnostics().Count(d => d.Severity == DiagnosticSeverity.Error);

        return new SourceMapFileEntry
        {
            RelativePath = source.RelativePath,
            LastWriteUtcTicks = source.LastWriteUtcTicks,
            SizeBytes = source.Length,
            LineCount = lineCount,
            NamespaceCount = namespaces.Count,
            Namespaces = namespaces,
            TypeCount = types.Count,
            Types = types,
            ParseErrorCount = diagnostics
        };
    }

    private static SourceMapTypeEntry CreateTypeEntry(SyntaxTree syntaxTree, TypeDeclarationSyntax typeDecl)
    {
        var methodEntries = new List<SourceMapMethodEntry>();
        methodEntries.AddRange(typeDecl.Members.OfType<ConstructorDeclarationSyntax>().Select(c => CreateConstructorEntry(syntaxTree, c)));
        methodEntries.AddRange(typeDecl.Members.OfType<MethodDeclarationSyntax>().Select(m => CreateMethodEntry(syntaxTree, m)));

        var propertyEntries = typeDecl.Members
            .OfType<PropertyDeclarationSyntax>()
            .Select(p => CreatePropertyEntry(syntaxTree, p))
            .ToList();

        var fieldEntries = typeDecl.Members
            .OfType<FieldDeclarationSyntax>()
            .SelectMany(f => CreateFieldEntries(syntaxTree, f))
            .ToList();

        var eventEntries = new List<SourceMapEventEntry>();
        eventEntries.AddRange(typeDecl.Members.OfType<EventFieldDeclarationSyntax>().SelectMany(e => CreateEventFieldEntries(syntaxTree, e)));
        eventEntries.AddRange(typeDecl.Members.OfType<EventDeclarationSyntax>().Select(e => CreateEventEntry(syntaxTree, e)));

        var namespaceName = typeDecl.Ancestors()
            .OfType<BaseNamespaceDeclarationSyntax>()
            .Select(static n => n.Name.ToString())
            .FirstOrDefault() ?? string.Empty;
        var qualifiedName = string.IsNullOrWhiteSpace(namespaceName)
            ? typeDecl.Identifier.ValueText
            : $"{namespaceName}.{typeDecl.Identifier.ValueText}";
        var baseTypes = typeDecl.BaseList is null
            ? new List<string>()
            : typeDecl.BaseList.Types.Select(static t => t.Type.ToString()).ToList();
        var modifiers = typeDecl.Modifiers.Select(static m => m.Text).ToList();
        var (startLine, endLine) = GetLineRange(syntaxTree, typeDecl);
        var defaultAccessibility = typeDecl.Parent is BaseNamespaceDeclarationSyntax || typeDecl.Parent is CompilationUnitSyntax
            ? "internal"
            : "private";
        return new SourceMapTypeEntry
        {
            Name = typeDecl.Identifier.ValueText,
            Namespace = namespaceName,
            QualifiedName = qualifiedName,
            Kind = typeDecl.Keyword.ValueText,
            Accessibility = GetAccessibility(typeDecl.Modifiers, defaultAccessibility),
            Modifiers = modifiers,
            BaseTypes = baseTypes,
            StartLine = startLine,
            EndLine = endLine,
            MethodCount = methodEntries.Count,
            PropertyCount = propertyEntries.Count,
            FieldCount = fieldEntries.Count,
            EventCount = eventEntries.Count,
            Methods = methodEntries,
            Properties = propertyEntries,
            Fields = fieldEntries,
            Events = eventEntries
        };
    }

    private static SourceMapMethodEntry CreateConstructorEntry(SyntaxTree syntaxTree, ConstructorDeclarationSyntax ctor)
    {
        var parameters = ctor.ParameterList.Parameters.Select(FormatParameter).ToList();
        var modifiers = ctor.Modifiers.Select(static m => m.Text).ToList();
        var (startLine, endLine) = GetLineRange(syntaxTree, ctor);
        return new SourceMapMethodEntry
        {
            Name = ctor.Identifier.ValueText,
            Kind = "constructor",
            ReturnType = ctor.Identifier.ValueText,
            Accessibility = GetAccessibility(ctor.Modifiers, "private"),
            Modifiers = modifiers,
            ParameterCount = parameters.Count,
            Parameters = parameters,
            Signature = $"{ctor.Identifier.ValueText}({string.Join(", ", parameters)})",
            StartLine = startLine,
            EndLine = endLine
        };
    }

    private static SourceMapMethodEntry CreateMethodEntry(SyntaxTree syntaxTree, MethodDeclarationSyntax method)
    {
        var parameters = method.ParameterList.Parameters.Select(FormatParameter).ToList();
        var typeParameters = method.TypeParameterList is null
            ? string.Empty
            : method.TypeParameterList.ToString();
        var name = method.Identifier.ValueText + typeParameters;
        var modifiers = method.Modifiers.Select(static m => m.Text).ToList();
        var (startLine, endLine) = GetLineRange(syntaxTree, method);

        return new SourceMapMethodEntry
        {
            Name = method.Identifier.ValueText,
            Kind = "method",
            ReturnType = method.ReturnType.ToString(),
            Accessibility = GetAccessibility(method.Modifiers, "private"),
            Modifiers = modifiers,
            ParameterCount = parameters.Count,
            Parameters = parameters,
            Signature = $"{method.ReturnType} {name}({string.Join(", ", parameters)})",
            StartLine = startLine,
            EndLine = endLine
        };
    }

    private static SourceMapPropertyEntry CreatePropertyEntry(SyntaxTree syntaxTree, PropertyDeclarationSyntax property)
    {
        var modifiers = property.Modifiers.Select(static m => m.Text).ToList();
        var accessors = property.AccessorList?.Accessors ?? default;
        var hasGetter = accessors.Any(static a => a.IsKind(SyntaxKind.GetAccessorDeclaration));
        var hasSetter = accessors.Any(static a => a.IsKind(SyntaxKind.SetAccessorDeclaration) || a.IsKind(SyntaxKind.InitAccessorDeclaration));
        var (startLine, endLine) = GetLineRange(syntaxTree, property);
        var getterText = hasGetter ? " get;" : string.Empty;
        var setterText = hasSetter ? " set;" : string.Empty;

        return new SourceMapPropertyEntry
        {
            Name = property.Identifier.ValueText,
            Type = property.Type.ToString(),
            Accessibility = GetAccessibility(property.Modifiers, "private"),
            Modifiers = modifiers,
            HasGetter = hasGetter,
            HasSetter = hasSetter,
            Signature = $"{property.Type} {property.Identifier.ValueText} {{{getterText}{setterText} }}".Replace("  ", " ").Trim(),
            StartLine = startLine,
            EndLine = endLine
        };
    }

    private static List<SourceMapFieldEntry> CreateFieldEntries(SyntaxTree syntaxTree, FieldDeclarationSyntax field)
    {
        var modifiers = field.Modifiers.Select(static m => m.Text).ToList();
        var (startLine, endLine) = GetLineRange(syntaxTree, field);
        return field.Declaration.Variables
            .Select(v => new SourceMapFieldEntry
            {
                Name = v.Identifier.ValueText,
                Type = field.Declaration.Type.ToString(),
                Accessibility = GetAccessibility(field.Modifiers, "private"),
                Modifiers = modifiers,
                Signature = $"{field.Declaration.Type} {v.Identifier.ValueText}",
                StartLine = startLine,
                EndLine = endLine
            })
            .ToList();
    }

    private static List<SourceMapEventEntry> CreateEventFieldEntries(SyntaxTree syntaxTree, EventFieldDeclarationSyntax eventField)
    {
        var modifiers = eventField.Modifiers.Select(static m => m.Text).ToList();
        var (startLine, endLine) = GetLineRange(syntaxTree, eventField);
        return eventField.Declaration.Variables
            .Select(v => new SourceMapEventEntry
            {
                Name = v.Identifier.ValueText,
                Type = eventField.Declaration.Type.ToString(),
                Accessibility = GetAccessibility(eventField.Modifiers, "private"),
                Modifiers = modifiers,
                Signature = $"event {eventField.Declaration.Type} {v.Identifier.ValueText}",
                StartLine = startLine,
                EndLine = endLine
            })
            .ToList();
    }

    private static SourceMapEventEntry CreateEventEntry(SyntaxTree syntaxTree, EventDeclarationSyntax eventDecl)
    {
        var modifiers = eventDecl.Modifiers.Select(static m => m.Text).ToList();
        var (startLine, endLine) = GetLineRange(syntaxTree, eventDecl);
        return new SourceMapEventEntry
        {
            Name = eventDecl.Identifier.ValueText,
            Type = eventDecl.Type.ToString(),
            Accessibility = GetAccessibility(eventDecl.Modifiers, "private"),
            Modifiers = modifiers,
            Signature = $"event {eventDecl.Type} {eventDecl.Identifier.ValueText}",
            StartLine = startLine,
            EndLine = endLine
        };
    }

    private static string FormatParameter(ParameterSyntax parameter)
    {
        var type = parameter.Type?.ToString() ?? "var";
        var modifier = parameter.Modifiers.Count > 0
            ? string.Join(" ", parameter.Modifiers.Select(static m => m.Text)) + " "
            : string.Empty;
        var defaultValue = parameter.Default is null ? string.Empty : $" = {parameter.Default.Value}";
        return $"{modifier}{type} {parameter.Identifier.ValueText}{defaultValue}".Trim();
    }

    private static int CountLines(string text)
    {
        if (text.Length == 0)
        {
            return 0;
        }

        var lineCount = 1;
        foreach (var ch in text)
        {
            if (ch == '\n')
            {
                lineCount++;
            }
        }

        return lineCount;
    }

    private static (int StartLine, int EndLine) GetLineRange(SyntaxTree syntaxTree, SyntaxNode node)
    {
        var span = syntaxTree.GetLineSpan(node.Span);
        return (span.StartLinePosition.Line + 1, span.EndLinePosition.Line + 1);
    }

    private static string GetAccessibility(SyntaxTokenList modifiers, string defaultAccessibility = "private")
    {
        if (modifiers.Any(static m => m.IsKind(SyntaxKind.PublicKeyword)))
        {
            return "public";
        }

        if (modifiers.Any(static m => m.IsKind(SyntaxKind.PrivateKeyword)))
        {
            return "private";
        }

        if (modifiers.Any(static m => m.IsKind(SyntaxKind.ProtectedKeyword))
            && modifiers.Any(static m => m.IsKind(SyntaxKind.InternalKeyword)))
        {
            return "protected internal";
        }

        if (modifiers.Any(static m => m.IsKind(SyntaxKind.ProtectedKeyword)))
        {
            return "protected";
        }

        if (modifiers.Any(static m => m.IsKind(SyntaxKind.InternalKeyword)))
        {
            return "internal";
        }

        return defaultAccessibility;
    }
